    // ...
];

$roster->getOldestStudent(); // John


class Roster {
    public function getOldestStudent() {
        $oldest = null;
        foreach ($this->students as $student) {
            if (!$oldest || $student->getAge() > $oldest->getAge()) {
                $oldest = $student;
            }
        }

        return $oldest;
    }
}


public function getOldestStudent() {
    return array_reduce($this->students, function ($oldest, $student) {
        if ($student->getAge() > $oldest->getAge()) {
            $oldest = $student;
        };

        return $oldest;
    }, reset($this->students));
}


public function getOldestStudent() {
    return array_reduce($this->students, function ($oldest, $student) {
        return $student->getAge() > $oldest->getAge() ? $student : $oldest;
    }, reset($this->students));
}
