foreach (range(1,100) as $num) {
    $output = '';

    if (0 === $num % 3) {
        $output .= 'Fizz';
    }

    if (0 === $num % 5) {
        $output .= 'Buzz';
    }

    if (!$output) {
        $output = $num;
    }

    echo $output . "\n";
}

class HArray extends \ArrayObject
{
    private $array = [];
    public function __construct(array $a = []) {
        $this->array = $a;
    }
    public function map(callable $func) {
        return new HArray(array_map($func, $this->array));
    }
    public function filter(callable $func = null) {
        return new HArray(array_filter($this->array, $func));
    }
    public function reduce(callable $func, $initial = null) {
        return array_reduce($this->array, $func, $initial);
    }
}

Why Arrays?

/** @param User[]|User|null $users */
function findUserGroups($users) {
    if ($users === null) return null;

    if ($users instanceof User) {
        $groups = Group::findByUser($users->id);
        return $groups;
    }

    if (count($users) === 0) {
        return [];
    } else {
        $groups = [];
        foreach ($users as $user) {
            $userGroups = Group::findByUser($user->id);
            if (!$groups) continue;
            $groups = array_merge($groups, $userGroups);
        }
    }

    return $groups;
}
